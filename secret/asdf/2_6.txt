#include <iostream>
#include <string>
using namespace std;
void obrni(string p1[], string p2[],int n) {
	for (int i = 0; i < n; i++)
	{
		p2[(n-1) - i] = p1[i];
	}
}
void getpolje(string polje[], int n) {
	for (int i = 0; i < n; i++)
	{
		cout << polje[i] << endl;
	}
}
int main() {
	int n = 3;
	string polje1[3] = {"jeda","sa","wew"};
	string polje2[3];

	obrni(polje1, polje2, n);

	getpolje(polje1, n);
	getpolje(polje2, n);

	return 0;
}
