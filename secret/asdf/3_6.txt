#include <iostream>
#include <stdio.h>
#include <algorithm>
#include <vector>
using namespace std;
bool postoji(vector<int>* polje, int n) {
	for (int i = 0; i < polje->size(); i++)
	{
		int m = polje->at(i);
		if (m == n)
			return true;
	}return false;
}
void getneg(int p1[],int p1b,int p2[],int p2b,vector<int>* v) {
	for (int i = 0; i < p1b; i++) { if ((p1[i] < 0) && (!postoji(v, p1[i]))) { v->push_back(p1[i]); } }
	for (int i = 0; i < p2b; i++) { if ((p2[i] < 0) && (!postoji(v, p2[i]))) { v->push_back(p2[i]); }	}
}
int main() {
	int polje1[6] = {4,6,-2,3,-7,-4};
	int polje2[4] = {3,-7,-1,-4};

	vector<int> vektor;
	vector<int>* v=&vektor;
	getneg(polje1, 6, polje2, 4, v);
	sort(vektor.begin(), vektor.end());
	unique(vektor.begin(), vektor.end());
	for (int i = 0; i < vektor.size(); i++)
	{
		cout << vektor[i] << endl;
	}

	return 0;
}
