#include <iostream>
#include <fstream>
#include <string>

using namespace std;
struct adresa {
	string ulica;
	int broj;
	int pb;
};
int main() {
	
	adresa poljea[3];
	poljea[0].ulica = "ulica prva";
	poljea[0].broj = 121;
	poljea[0].pb = 10000;
	poljea[1].ulica = "ulica druga";
	poljea[1].broj = 73;
	poljea[1].pb = 10000;
	poljea[2].ulica = "ulica treca";
	poljea[2].broj = 49;
	poljea[2].pb = 10000;

	ofstream dat("adrese.txt");
	for (int i = 0; i < 3; i++)
	{
		dat << poljea[i].ulica<<","<<poljea[i].broj << ","<<poljea[i].pb << endl;
	}


	dat.close();

	return 0;
}
