#include <iostream>
#include <fstream>
#include <string>
#include <vector>
using namespace std;
bool prost(int n)
{
	for (int i = 2; i <= n / 2; i++) {
		if (n%i == 0) {
			return false;
		}
	}
	return true;
}
bool postoji(vector<int>* polje,int n) {
	for (int i = 0; i < polje->size(); i++)
	{
		int m = polje->at(i);
		if (m == n)
			return true;
	}return false;
}
int main() {
	int n = 100;
	int van;
	vector<int> brojevi;
	ifstream dat("brojevi.txt");
	while (dat>>van)
	{
		brojevi.push_back(van);
		cout << van << endl;
	}
	vector<int> prosti;
	vector<int>* pp=&prosti;
	ofstream dat2("prosti.txt");
	for (int i = 2; i < brojevi.size(); i++)
	{
		if (prost(brojevi[i])) {
			dat2 << brojevi[i] << endl;
			prosti.push_back(brojevi[i]);
		}
	}
	vector<int> neparni;
	ofstream dat3("neparni.txt");
	for (int i = 0; i < brojevi.size(); i++)
	{
		if (brojevi[i] % 2 != 0) {
			dat3 << brojevi[i] << endl;
			neparni.push_back(brojevi[i]);
		}
	}
	vector<int> nepbezpro;
	ofstream dat4("neparnibezprostih.txt");
	for (int i = 0; i < neparni.size(); i++)
	{
		if (!postoji(pp, neparni[i])) {
		nepbezpro.push_back(neparni[i]);
		dat4 << neparni[i] << endl;
		}
	}

	dat.close();
	dat2.close();
	dat3.close();
	dat4.close();
	
	return 0;
}
