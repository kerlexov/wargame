#include <iostream>
#include <string>
using namespace std;

int main() {
	int n;
	cout << "koliko unest" << endl;
	cin >> n;
	string* mem = new string[n+1];
	for (int i = 0; i <= n; i++)
	{
		getline(cin, mem[i]);
	}
	for (int i = n; i >= 0; i--)
	{
		cout << mem[i] << endl;
	}


	delete[] mem;
	return 0;
}
